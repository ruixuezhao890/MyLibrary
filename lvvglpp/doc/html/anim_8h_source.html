142 template <class T> void set_exec_cb(ExecCb<T> exec_cb) {
143 // the exec_cb callback takes a void* as 1st parameter, in order to
144 // accommodate pointers of any type; this means the animation object
145 // isn't passed to the callback; to make the callback accept a reference
146 // to a C++ class instance, we need to be able to pass a reference
147 // to the callback function somehow. For callbacks with lv_anim_t* first
148 // params, it's easy: set user_data to this and retrieve this in callback.
149 // For this one, this is trickier. My approach is to provide a callback
150 // container that also stores the variable that would normally be passed
151 // to exec_cb. Then we change the variable in the lv_anim_t object to
152 // a pointer to the callback container. We can then access the callback
153 // and the variable upon which it acts.
154 lv_anim_set_var(this->raw_ptr(), static_cast<void*>(this));
155 using ExecCbType = Callback<void, T&, int32_t>;
156 this->exec_cb = std::make_shared<ExecCbType>(exec_cb);
157 // definition of callback wrapper function; this will call C++ callback
158 // with stored var
159 auto f = [](void * lv_obj, int32_t value) {
160 auto anim = reinterpret_cast<Animation*>(lv_obj);
161 auto cb = *std::dynamic_pointer_cast<ExecCbType>(anim->exec_cb);
162 cb(anim->get_var<T>(), value);
163 };
164 lv_anim_set_exec_cb(this->raw_ptr(), f);
165 }
