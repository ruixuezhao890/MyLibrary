51 static RawDblArray<char> str_vector_to_char_array(const std::vector<std::string> & arr) {
52 auto raw_arr = make_double_array<char>(arr.size());
53 for (size_t n=0; n<arr.size(); n++) {
54 raw_arr.get()[n] = static_cast<char*>(calloc(arr[n].size()+1, sizeof(char)));
55 memset(raw_arr.get()[n], 0, arr[n].size()+1);
56 std::copy(arr[n].begin(), arr[n].end(), raw_arr.get()[n]);
57 }
58 return raw_arr;
